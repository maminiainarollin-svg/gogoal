// --- LOGIQUE DE CALCUL CORRIGÉE --- function calculateStats() { const stats = new Array(rows * cols).fill(0.5); // 50% = Inconnu const counts = new Array(rows * cols).fill(null).map(() => ({ b: 0, m: 0 })); history.forEach((game, idx) => { const weight = Math.pow(0.95, idx); // Les parties récentes comptent plus const data = game.data || game.cells; data.forEach((val, i) => { if (val === 1) counts[i].b += weight; // Ballon enregistré if (val === 2) counts[i].m += weight; // Mine enregistrée }); }); return counts.map(c => { const total = c.b + c.m; if (total === 0) return 0.5; // Case jamais testée // LOGIQUE DE SÉCURITÉ : // On prend les ballons et on SOUSTRAIT fortement l'impact des mines. // Si une case a 1 ballon et 1 mine, son score sera bas (danger). const score = (c.b - (c.m * 2)) / (c.b + c.m + 1); // On normalise entre 0 et 1 (0 = Danger mortel, 1 = Très sûr) return Math.max(0.1, Math.min(0.9, 0.5 + score)); }); } function analyze() { const stats = calculateStats(); let msg = "⚽ CHEMIN CONSEILLÉ (SÉCURITÉ) :\n\n"; for (let c = 0; c < cols; c++) { let bestR = 0; let maxSafety = -1; for (let r = 0; r < rows; r++) { let idx = r * cols + c; // On cherche la case avec le score de sécurité le plus élevé if (stats[idx] > maxSafety) { maxSafety = stats[idx]; bestR = r + 1; } } const dangerPercent = Math.round((1 - maxSafety) * 100); msg += `Etape ${c+1} : Ligne ${bestR} (Risque estimé: ${dangerPercent}%)\n`; } alert(msg); }
